module Main where

fib :: Int -> Int
fib 0 = 0
fib 1 = 1
fib n = fib (n - 1) + fib (n - 2)

main :: IO ()
main = do
  let values = map fib [0..15]
  putStrLn $ "fib(0..15) = " ++ show values
